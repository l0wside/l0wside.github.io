261 VLC_USED
262 static inline const char *config_CategoryNameGet( int i_value )
263 {
264  int i = 0;
265  while( categories_array[i].psz_name != NULL )
266  {
267  if( categories_array[i].i_id == i_value )
268  {
269  return vlc_gettext(categories_array[i].psz_name);
270  }
271  i++;
272  }
273  return NULL;
274 }
